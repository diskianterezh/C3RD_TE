c3rd_other_events.1 = { #Empire leader has too low martial in a war
    type = country_event
    title = "c3rd_other_events.1.t"
    desc = {
        desc = "c3rd_other_events.1.desc"
        triggered_desc = {
            trigger = {
                any_country_culture = {
                    is_integrated = yes
                    this.culture.culture_group = culture_group:hellenic
                }
            }
            desc = "c3rd_other_events.1.desc.1"
            
        }
    }
 
    picture = successors

    trigger = {
        has_land = yes
        current_date >= 700.1.1
        is_monarchy = yes
        OR = {
            government = imperium
            country_culture_group = hellenic
        }
        rank = great_power
        current_ruler = {
            martial < 9
        }
        NOR = {
            has_country_modifier = unworthy_leader_1
            has_country_modifier = unworthy_leader_2
            has_country_modifier = unworthy_leader_3
        }
    }


    immediate = {
        if = {
            limit = {
                current_date >= 1050.1.1 
            }
            add_country_modifier = {
                name = unworthy_leader_3
                duration = 7300
            }
        }
        else_if = {
            limit = {
                current_date >= 900.1.1 
            }
            add_country_modifier = {
                name = unworthy_leader_2
                duration = 7300
            }
        }
        else = {
            add_country_modifier = {
                name = unworthy_leader_1
                duration = 7300
            }
        }
        if = {
            limit = {
                country_culture_group = hellenic
            }
            add_legitimacy = -10
        }
        add_stability = -5
           
    }

    option = { #please help me improve
        name = c3rd_other_events.1.a
        custom_tooltip = c3rd_other_events.1.a.tt1
        trigger = {
            any_character = {
                has_any_military_office_trigger = yes
                martial > root.current_ruler.martial
            }
        }
        highlight = yes
        add_legitimacy = -25
        current_ruler = {
            add_martial = 1
        }
    }
    option = { #ok, don't care
        name = c3rd_other_events.1.b
        custom_tooltip = c3rd_other_events.1.b.tt1
        current_ruler = {
            add_popularity = 10
            add_prominence = 10
        }
        hidden_effect = {
            every_country_culture = {
                limit = {
                    this.culture.culture_group = culture_group:hellenic
                }
                add_country_culture_modifier = {
                    name = "unworthy_ruler_tradition"
                    duration = 3650
                }
            }
        }
    }
    option = { #i'm pretty smooth
        name = c3rd_other_events.1.c
        custom_tooltip = c3rd_other_events.1.c.tt1
        trigger = {
            current_ruler = {
                popularity >= 50
                charisma >= 10
            }
        }
        highlight = yes
        add_legitimacy = 10
    }
}

c3rd_other_events.4 = { #social crisis : too few integrated pops
    type = country_event
    title = "c3rd_other_events.4.t"
    desc = "c3rd_other_events.4.desc"
 
    picture = persia_marching

    trigger = {
        has_land = yes
        current_date >= 700.1.1
        rank = great_power
        OR = {
            AND = {
                country_culture_group = latin
                country_population > noble_country_check_italian
            }
            country_population > noble_country_check
        }
        is_ai = no
        NOR = {
            has_country_modifier = administrative_difficulty
            has_country_modifier = administrative_difficulty_small
        }
    }

    weight_multiplier = {
		modifier = {
			factor = 2
			country_population > noble_country_check_a_lot
		}
		modifier = {
			factor = 1.5
			stability <= 40
		}
		modifier = {
			factor = 2
			stability <= 25
		}
		modifier = {
			factor = 0.5
			stability >= 60
		}
    }



    immediate = {
        ordered_country_culture = {
            limit = {
                NOT = {
                    has_pop_type_right = nobles
                }
            }
            max = 1
            order_by = country_culture_pop_count
            save_scope_as = chosen_new_nobles
        }
    }

    option = {
        name = c3rd_other_events.4.a
        custom_tooltip = c3rd_other_events.4.a.tt1
        culture_state_loyalty_effect_c3rd = yes
        scope:chosen_new_nobles = {
            set_country_culture_right = nobles
            add_country_culture_modifier = {
                name = "new_culture_elite"
                duration = 1825
            }
            add_country_culture_modifier = {
                name = "promoted_administrators_in_subcultures"
                duration = -1
            }
			custom_tooltip = culture_group_boosted_c3rd_tt
			hidden_effect = {
				root={every_country_culture = {
					limit = {
						this.culture.culture_group = scope:chosen_new_nobles.culture.culture_group
						NOT = { this = scope:chosen_new_nobles }
						NOT = { has_country_culture_modifier = administrators_cultmodifier }
					}
					add_country_culture_modifier = {
						name = "administrators_cultmodifier"
						duration = -1
					}
				}}
			}
            
        }
    }
    option = {
        name = c3rd_other_events.4.b
        custom_tooltip = c3rd_other_events.4.b.tt1
        scope:chosen_new_nobles = {
            add_country_culture_modifier = {
                name = "new_elite_refused"
                duration = 1825
            }
        }
        add_country_modifier = {
            name = administrative_difficulty
            duration = -1
        }
    }
    option = {
        name = c3rd_other_events.4.c
        custom_tooltip = c3rd_other_events.4.c.tt1
        highlight = yes
        trigger = {
            current_ruler = {
                OR = {
                    finesse >= 10
                    has_trait = crafty
                    has_trait = intelligent
                }
            }
            NOT = {has_country_modifier = administrative_difficulty_small}
        }
        add_treasury = {
            value = has_monthly_income
            multiply = -5
        }
        add_political_influence = -25
        add_country_modifier = {
            name = administrative_difficulty_small
            duration = 3650
        }
    }
}
