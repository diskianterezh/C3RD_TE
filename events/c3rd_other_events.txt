c3rd_other_events.1 = { #Empire leader has too low martial in a war
    type = country_event
    title = "c3rd_other_events.1.t"
    desc = {
        desc = "c3rd_other_events.1.desc"
        triggered_desc = {
            trigger = {
                any_country_culture = {
                    is_integrated = yes
                    this.culture.culture_group = culture_group:hellenic
                }
            }
            desc = "c3rd_other_events.1.desc.1"

        }
    }

    picture = successors

    trigger = {
        has_land = yes
        current_date >= 700.1.1
        is_monarchy = yes
        OR = {
            government = imperium
            country_culture_group = hellenic
        }
        rank = great_power
        current_ruler = {
            martial < 9
        }
        NOR = {
            has_country_modifier = unworthy_leader_1
            has_country_modifier = unworthy_leader_2
            has_country_modifier = unworthy_leader_3
        }
    }


    immediate = {
        if = {
            limit = {
                current_date >= 1050.1.1
            }
            add_country_modifier = {
                name = unworthy_leader_3
                duration = 7300
            }
        }
        else_if = {
            limit = {
                current_date >= 900.1.1
            }
            add_country_modifier = {
                name = unworthy_leader_2
                duration = 7300
            }
        }
        else = {
            add_country_modifier = {
                name = unworthy_leader_1
                duration = 7300
            }
        }
        if = {
            limit = {
                country_culture_group = hellenic
            }
            add_legitimacy = -10
        }
        add_stability = -5

    }

    option = { #please help me improve
        name = c3rd_other_events.1.a
        custom_tooltip = c3rd_other_events.1.a.tt1
        trigger = {
            any_character = {
                has_any_military_office_trigger = yes
                martial > root.current_ruler.martial
            }
        }
        highlight = yes
        add_legitimacy = -25
        current_ruler = {
            add_martial = 1
        }
    }
    option = { #ok, don't care
        name = c3rd_other_events.1.b
        custom_tooltip = c3rd_other_events.1.b.tt1
        current_ruler = {
            add_popularity = 10
            add_prominence = 10
        }
        hidden_effect = {
            every_country_culture = {
                limit = {
                    this.culture.culture_group = culture_group:hellenic
                }
                add_country_culture_modifier = {
                    name = "unworthy_ruler_tradition"
                    duration = 3650
                }
            }
        }
    }
    option = { #i'm pretty smooth
        name = c3rd_other_events.1.c
        custom_tooltip = c3rd_other_events.1.c.tt1
        trigger = {
            current_ruler = {
                popularity >= 50
                charisma >= 10
            }
        }
        highlight = yes
        add_legitimacy = 10
    }
}

c3rd_other_events.4 = { #social crisis : too few integrated pops
    type = country_event
    title = "c3rd_other_events.4.t"
    desc = "c3rd_other_events.4.desc"

    picture = persia_marching

    trigger = {
        has_land = yes
        current_date >= 700.1.1
        rank = great_power
        OR = {
            AND = {
                country_culture_group = latin
                country_population > noble_country_check_italian
            }
            country_population > noble_country_check
        }
        is_ai = no
        NOR = {
            has_country_modifier = administrative_difficulty
            has_country_modifier = administrative_difficulty_small
        }
    }

    weight_multiplier = {
		modifier = {
			factor = 2
			country_population > noble_country_check_a_lot
		}
		modifier = {
			factor = 1.5
			stability <= 40
		}
		modifier = {
			factor = 2
			stability <= 25
		}
		modifier = {
			factor = 0.5
			stability >= 60
		}
    }



    immediate = {
        ordered_country_culture = {
            limit = {
                NOT = {
                    has_pop_type_right = nobles
                }
            }
            max = 1
            order_by = country_culture_pop_count
            save_scope_as = chosen_new_nobles
        }
    }

    option = {#new elite
        name = c3rd_other_events.4.a
        custom_tooltip = c3rd_other_events.4.a.tt1
        culture_state_loyalty_effect_c3rd = yes
        scope:chosen_new_nobles = {
            set_country_culture_right = nobles
            add_country_culture_modifier = {
                name = "new_culture_elite"
                duration = 1825
            }
            add_country_culture_modifier = {
                name = "promoted_administrators_in_subcultures"
                duration = -1
            }
			custom_tooltip = culture_group_boosted_c3rd_tt
			hidden_effect = {
				root={every_country_culture = {
					limit = {
						this.culture.culture_group = scope:chosen_new_nobles.culture.culture_group
						NOT = { this = scope:chosen_new_nobles }
						NOT = { has_country_culture_modifier = administrators_cultmodifier }
					}
					add_country_culture_modifier = {
						name = "administrators_cultmodifier"
						duration = -1
					}
				}}
			}

        }
    }
    option = {#refuse
        name = c3rd_other_events.4.b
        custom_tooltip = c3rd_other_events.4.b.tt1
        scope:chosen_new_nobles = {
            add_country_culture_modifier = {
                name = "new_elite_refused"
                duration = 1825
            }
        }
        add_country_modifier = {
            name = administrative_difficulty
            duration = -1
        }
    }
    option = {#half
        name = c3rd_other_events.4.c
        custom_tooltip = c3rd_other_events.4.c.tt1
        highlight = yes
        trigger = {
            current_ruler = {
                OR = {
                    finesse >= 10
                    has_trait = crafty
                    has_trait = intelligent
                }
            }
            NOT = {has_country_modifier = administrative_difficulty_small}
        }
        add_treasury = {
            value = has_monthly_income
            multiply = -5
        }
        add_political_influence = -25
        add_country_modifier = {
            name = administrative_difficulty_small
            duration = 3650
        }
    }

### ADMINISTRATIVE EVENTS ####

### Basically nobles (great families) taking more power in the realm
### Nothing will almost ever happen as long as you don't reform the administrative law, or enact the Solidus
### Constantine reform is cool because it reduce military anarchy, however it will progressively give power to the governors
### Everything will although stay minor UNTIL you enact Solidus - and then things will snowball
### Solidus is really great because it delete all the monetary crisis mechanics so you don't have to deal with that anymore, but with a secondary effect.
### Those Administrative things are the secondary effect





c3rd_other_events.80 = { #Notification of governorship taken by family
	type = minor_country_event
	title= "c3rd_other_events.80.t"
	desc = "c3rd_other_events.80.desc"
	interface_lock = no
	
	left_portrait = scope:autonomy_governorship.var:governorship_owner
	
	trigger = {
		current_date > 750.1.1
        scope:autonomy_governorship = {
            exists = governor
            governor = {
                is_ruler = no
            }
            NOR = {
				AND = {
					has_variable = governorship_owner
                    var:governorship_owner = {
                        is_head_of_family = yes
                        family = scope:autonomy_governorship.governor.family
                    }
				}
				AND = {
					has_variable = governorship_owner
					var:governorship_owner = governor
				}
            }
			OR = {
				AND = {
					region_control_value >= 30
					root={has_law = decentralized_administration_law}
				}
				AND = {
					region_control_value >= 40
					root={has_country_modifier = solidus_introduced}
				}
				region_control_value >= 50
			}  
        }
	}
	
	immediate = {
        scope:autonomy_governorship = {
            if = {
                limit = {
                    has_variable = governorship_owner
                }
                var:governorship_owner = {
                    save_scope_as = ex_owner
                }
                remove_variable = governorship_owner
                scope:ex_owner = {
                    recalculate_ownership_char = yes
                }
            }
            set_variable = {
                name = governorship_owner
                value = governor
            }
            if = {
                limit = {
                    has_variable = governorship_owner_ratio
                }
                remove_variable = governorship_owner_ratio
            }
            set_variable = { #equivalent  appeler calculate_the_gov_ownership = yes
                name = governorship_owner_ratio
                value = region_control_value #prend le gov alors que calculate_the_gov_ownership prend le owner, ce qui est la mme chose ici
            }
            hidden_effect = {
                governor = {
                    if = {
                        limit = {
                            has_variable = country_ownership
                        }
                        remove_variable = country_ownership
                    }
                    save_scope_as = influent_character
                    set_variable = {
                        name = country_ownership
                        value = country_region_control
                    }
                    if = {
                        limit = { 
                            country_region_control >= 25
                            NOT = {
                                has_character_modifier = major_influence_aristocrat
                            }
                        }
                        add_character_modifier = {
                            name = major_influence_aristocrat
                            duration = -1
                        }
                    }
                    if = {
                        limit = { 
                            has_character_modifier = major_influence_aristocrat
                            country_region_control < 20
                        }
                        remove_character_modifier = major_influence_aristocrat
                    }
                }
            }
        }

	}

	option = {
		name = "c3rd_other_events.80.a"
        if = {
            limit = {
                scope:influent_character = {
                    is_minor_character = no
                }
            }
            custom_tooltip = c3rd_other_events.80.a.tt1
        }
        else = {
            custom_tooltip = c3rd_other_events.80.a.tt2
        }
	}
}
